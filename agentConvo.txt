```python
# Simple To-Do List App

# Initialize an empty list to store tasks
todo_list = []

def add_task(task):
    """Add a new task to the to-do list."""
    todo_list.append(task)
    print(f'"{task}" has been added to your to-do list.')

def delete_task(task_number):
    """Delete a task from the to-do list by its number."""
    if 0 <= task_number < len(todo_list):
        removed_task = todo_list.pop(task_number)
        print(f'"{removed_task}" has been removed from your to-do list.')
    else:
        print("Invalid task number.")

def view_tasks():
    """Display all tasks in the to-do list."""
    if not todo_list:
        print("Your to-do list is empty.")
    else:
        print("Your to-do list:")
        for idx, task in enumerate(todo_list):
            print(f"{idx + 1}. {task}")

def main():
    while True:
        print("\nChoose an option:")
        print("1. Add a task")
        print("2. Delete a task")
        print("3. View tasks")
        print("4. Exit")
        choice = input("Enter your choice (1-4): ")

        if choice == '1':
            task = input("Enter the task: ")
            add_task(task)
        elif choice == '2':
            view_tasks()
            try:
                task_num = int(input("Enter the task number to delete: ")) - 1
                delete_task(task_num)
            except ValueError:
                print("Please enter a valid number.")
        elif choice == '3':
            view_tasks()
        elif choice == '4':
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```

This script provides a simple command-line interface to manage your to-do list. You can add tasks, delete them by number, view all tasks, or exit the application. Would you like a version with a graphical user interface or integration with a database?
